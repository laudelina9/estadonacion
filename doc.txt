        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        function playBeep(type) {
            try {
                let osc = audioCtx.createOscillator();
                let gain = audioCtx.createGain();
                osc.connect(gain);
                gain.connect(audioCtx.destination);
                
                if (type === 'click') {
                    osc.frequency.setValueAtTime(350, audioCtx.currentTime);
                    gain.gain.setValueAtTime(0.08, audioCtx.currentTime);
                    osc.start(); osc.stop(audioCtx.currentTime + 0.06);
                } else if (type === 'success') {
                    osc.type = 'triangle';
                    osc.frequency.setValueAtTime(523.25, audioCtx.currentTime); 
                    osc.frequency.setValueAtTime(659.25, audioCtx.currentTime + 0.12); 
                    osc.frequency.setValueAtTime(783.99, audioCtx.currentTime + 0.24); 
                    gain.gain.setValueAtTime(0.12, audioCtx.currentTime);
                    osc.start(); osc.stop(audioCtx.currentTime + 0.45);
                } else if (type === 'alarm') {
                    osc.type = 'sawtooth';
                    osc.frequency.setValueAtTime(150, audioCtx.currentTime);
                    osc.frequency.linearRampToValueAtTime(300, audioCtx.currentTime + 0.3);
                    gain.gain.setValueAtTime(0.2, audioCtx.currentTime);
                    osc.start(); osc.stop(audioCtx.currentTime + 0.4);
                }
            } catch(e) { console.log("Audio no inicializado"); }
        }

        function requestStart() {
            const container = document.documentElement;
            const codeInput = document.getElementById("game-code").value.trim().toUpperCase();
            
            if (codeInput !== TEACHER_CODE && container.requestFullscreen) {
                container.requestFullscreen().then(startGame).catch(() => {
                    alert("Es mandatorio validar la pantalla completa para realizar la prueba.");
                });
            } else {
                startGame();
            }
        }

        function startGame() {
            if (localStorage.getItem("desafio_completado") === "true") {
                forceLockout();
                return;
            }

            const nameInput = document.getElementById("user-name").value.trim();
            const codeInput = document.getElementById("game-code").value.trim().toUpperCase();
            const errorDiv = document.getElementById("login-error");

            if (!nameInput || !codeInput) {
                errorDiv.style.display = "block";
                errorDiv.innerText = "Por favor, completá tu Nombre, Apellido y Código para continuar.";
                return;
            }

            if (codeInput === TEACHER_CODE) {
                isTeacherMode = true;
            } else if (STUDENT_CODES.includes(codeInput)) {
                isTeacherMode = false;
            } else {
                errorDiv.style.display = "block";
                errorDiv.innerText = "El código ingresado no figura en los registros de juego.";
                return;
            }

            studentName = nameInput;
            studentCode = codeInput;
            errorDiv.style.display = "none";
            gameActive = true;
            playBeep('click');

            document.getElementById("screen-login").classList.remove("active");
            document.getElementById("screen-game").classList.add("active");
            
            if (!isTeacherMode) {
                activateSecurityTramps();
            }
            showQuestion();
        }

        function activateSecurityTramps() {
            document.addEventListener('copy', e => e.preventDefault());
            document.addEventListener('paste', e => e.preventDefault());
            document.addEventListener('contextmenu', e => e.preventDefault());
            
            document.addEventListener('visibilitychange', () => {
                if (document.hidden && gameActive) {
                    triggerTrampleLockout("El estudiante minimizó, cambió de pestaña o dividió su pantalla principal.");
                }
            });

            document.addEventListener('fullscreenchange', () => {
                if (!document.fullscreenElement && gameActive) {
                    triggerTrampleLockout("El estudiante interrumpió voluntariamente el modo de pantalla completa.");
                }
            });
        }

        function triggerTrampleLockout(reason) {
            gameActive = false;
            playBeep('alarm');
            if(document.exitFullscreen) { document.exitFullscreen().catch(()=>{}); }

            localStorage.setItem("desafio_completado", "true");

            const payload = {
                "NOTA_FINAL_ALUMNO": "0.00 - EXAMEN ANULADO POR TRASGRESIÓN DE SEGURIDAD",
                estudiante: studentName,
                codigoUsado: studentCode,
                motivoCierre: reason,
                estadoSistema: "Cierre forzado automático"
            };

            fetch(FORMSPREE_URL, {
                method: "POST",
                headers: { "Content-Type": "application/json", "Accept": "application/json" },
                body: JSON.stringify(payload)
            }).finally(() => {
                document.getElementById("screen-game").classList.remove("active");
                document.getElementById("screen-login").classList.remove("active");
                const endScreen = document.getElementById("screen-end");
                document.getElementById("end-icon").innerText = "🚨";
                document.getElementById("end-content").innerHTML = `
                    <h2>Examen Anulado de Forma Automática</h2>
                    <p style="color: #f87171; font-weight: bold;">Se detectó una acción fraudulenta o de salida del entorno seguro.</p>
                    <p style="color: var(--text-muted);">Tu examen se ha cerrado con una calificación de 0.00 y las causas de la exclusión técnica ya se guardaron en los reportes de la profesora.</p>
                `;
                endScreen.classList.add("active");
            });
        }

            document.getElementById("btn-next").innerText = currentQuestionIndex === quizData.length - 1 ? "Finalizar 🏁" : "Avanzar ➡️";

            const imgUrl = TOPIC_IMAGES[qData.topic] || TOPIC_IMAGES["Estado"];

            const header = document.createElement("div");
            header.innerHTML = `
                <img src="${imgUrl}" class="topic-image" alt="Eje Temático - Argentina" onerror="this.src='https://placehold.co/800x240/1e1e24/6366f1?text=Ciencias+Politicas+Argentina'">
                <div class="icon-header">${qData.icon}</div>
                <span style="font-size:12px; text-transform:uppercase; font-weight:bold; color:var(--primary)">Tema: ${qData.topic} (${currentQuestionIndex+1}/15)</span>
                <h3 style="margin-top:8px; line-height: 1.4;">${qData.q}</h3>`;
            container.appendChild(header);

            if (qData.type === "choice") {
                renderChoice(qData, container);
            } else if (qData.type === "grid") {
                renderGrid(qData, container);
            } else if (qData.type === "match") {
                renderMatch(qData, container);
            }
        }

        function renderChoice(qData, container) {
            const wrapper = document.createElement("div");
            wrapper.className = "options-grid";
            qData.options.forEach((opt, idx) => {
                const card = document.createElement("div");
                card.className = "option-card";
                if (userAnswers[qData.id] === idx) card.classList.add("selected");
                card.innerText = opt;
                card.onclick = () => {
                    playBeep('click');
                    document.querySelectorAll(".option-card").forEach(c => c.classList.remove("selected"));
                    card.classList.add("selected");
                    userAnswers[qData.id] = idx;
                };
                wrapper.appendChild(card);
            });
            container.appendChild(wrapper);
        }

        function renderGrid(qData, container) {
            const table = document.createElement("table");
            table.className = "grid-table";
            table.innerHTML = `<tr><th>${qData.headers[0]}</th><th>${qData.headers[1]}</th></tr>`;

            qData.rows.forEach(row => {
                const tr = document.createElement("tr");
                const tdLabel = document.createElement("td");
                tdLabel.innerText = row.label;
                tdLabel.style.fontWeight = "600";

                const tdSelect = document.createElement("td");
                const select = document.createElement("select");
                row.options.forEach(opt => {
                    const o = document.createElement("option");
                    o.value = opt; o.innerText = opt;
                    if (userAnswers[qData.id] === opt) o.selected = true;
                    select.appendChild(o);
                });

                select.onchange = () => {
                    playBeep('click');
                    userAnswers[qData.id] = select.value;
                };
                tdSelect.appendChild(select);
                tr.appendChild(tdLabel); tr.appendChild(tdSelect);
                table.appendChild(tr);
            });
            container.appendChild(table);
        }

        let selectedLeft = null;
        function renderMatch(qData, container) {
            selectedLeft = null;
            const matchWrapper = document.createElement("div");
            matchWrapper.className = "matching-container";

            const colLeft = document.createElement("div");
            colLeft.className = "matching-column";
            const colRight = document.createElement("div");
            colRight.className = "matching-column";

            if (!userAnswers[qData.id]) userAnswers[qData.id] = {};

            qData.left.forEach(item => {
                const div = document.createElement("div");
                div.className = "match-item"; div.innerText = item;
                if (userAnswers[qData.id][item]) div.classList.add("linked");
                div.onclick = () => {
                    if (div.classList.contains("linked")) return;
                    playBeep('click');
                    document.querySelectorAll(".matching-column:first-child .match-item").forEach(i => i.classList.remove("selected"));
                    div.classList.add("selected");
                    selectedLeft = item;
                };
                colLeft.appendChild(div);
            });

            qData.right.forEach(item => {
                const div = document.createElement("div");
                div.className = "match-item"; div.innerText = item;
                Object.values(userAnswers[qData.id]).forEach(val => {
                    if (val === item) div.classList.add("linked");
                });
                div.onclick = () => {
                    if (div.classList.contains("linked") || !selectedLeft) return;
                    playBeep('click');
                    userAnswers[qData.id][selectedLeft] = item;
                    document.querySelector(`.matching-column:first-child .match-item.selected`).classList.add("linked");
                    div.classList.add("linked");
                    selectedLeft = null;
                    document.querySelectorAll(".match-item").forEach(i => i.classList.remove("selected"));
                };
                colRight.appendChild(div);
            });

            matchWrapper.appendChild(colLeft); matchWrapper.appendChild(colRight);
            
            const resetBtn = document.createElement("button");
            resetBtn.className = "btn btn-secondary";
            resetBtn.style.cssText = "margin-top:15px; font-size:12px; padding:8px 16px;";
            resetBtn.innerText = "♻️ Reestablecer uniones";
            resetBtn.onclick = () => {
                playBeep('click');
                userAnswers[qData.id] = {};
                renderMatch(qData, container);
            };

            container.appendChild(matchWrapper);
            container.appendChild(resetBtn);
        }

        function navigateQuestion(step) {
            playBeep('click');
            if (currentQuestionIndex === quizData.length - 1) {
                gameActive = false;
                processFinalResults();
            } else {
                currentQuestionIndex += step;
                showQuestion();
            }
        }

        function processFinalResults() {
            let correctCount = 0;
            let reportHTML = "";

            quizData.forEach(q => {
                const ans = userAnswers[q.id];
                let isCorrect = false;

                if (q.type === "choice") {
                    if (ans === q.correct) isCorrect = true;
                } else if (q.type === "grid") {
                    if (ans === q.rows[0].correct) isCorrect = true;
                } else if (q.type === "match") {
                    let allCorrect = true;
                    if (!ans || Object.keys(ans).length === 0) {
                        allCorrect = false;
                    } else {
                        Object.keys(q.pairs).forEach(key => {
                            if (ans[key] !== q.pairs[key]) allCorrect = false;
                        });
                    }
                    if (allCorrect) isCorrect = true;
                }

                if (isCorrect) correctCount++;
                reportHTML += `<li>Pregunta ${q.id} (${q.topic}): ${isCorrect ? "✅ Acertaste" : "❌ No acertaste"}</li>`;
            });

            let finalGrade = ((correctCount / quizData.length) * 10).toFixed(2);

            if (isTeacherMode) {
                if(document.exitFullscreen) { document.exitFullscreen().catch(()=>{}); }
                playBeep('success');
                document.getElementById("screen-game").classList.remove("active");
                const endScreen = document.getElementById("screen-end");
                document.getElementById("end-icon").innerText = "🛠️";
                document.getElementById("end-content").innerHTML = `
                    <h2>Simulación de Docente Finalizada</h2>
                    <p style="color: var(--secondary); font-size:18px; font-weight:bold;">Nota simulada: ${finalGrade} / 10</p>
                    <p style="color: var(--text-muted); font-size:14px;">La plataforma operó en entorno seguro aislado: No se consumió espacio de almacenamiento de Formspree.</p>
                    <div class="teacher-results-box">
                        <strong style="color: var(--primary);">Validación de aciertos de cada pantalla:</strong>
                        <ul style="margin: 10px 0; padding-left: 20px; font-size: 14px; line-height: 1.6;">
                            ${reportHTML}
                        </ul>
                    </div>
                    <p style="font-size:13px; color:var(--accent); margin-top:15px;">Podés reiniciar de manera ilimitada recargando la pestaña.</p>
                `;
                endScreen.classList.add("active");
                return;
            }

            if(document.exitFullscreen) { document.exitFullscreen().catch(()=>{}); }
            localStorage.setItem("desafio_completado", "true");

            const payload = {
                "NOTA_FINAL_ALUMNO": `${finalGrade} / 10.00`,
                estudiante: studentName,
                codigoUsado: studentCode,
                respuestasCorrectas: `${correctCount} de 15`,
                detalles: JSON.stringify(userAnswers)
            };

            fetch(FORMSPREE_URL, {
                method: "POST",
                headers: { "Content-Type": "application/json", "Accept": "application/json" },
                body: JSON.stringify(payload)
            }).finally(() => {
                playBeep('success');
                document.getElementById("screen-game").classList.remove("active");
                document.getElementById("screen-end").classList.add("active");
            });
        }
